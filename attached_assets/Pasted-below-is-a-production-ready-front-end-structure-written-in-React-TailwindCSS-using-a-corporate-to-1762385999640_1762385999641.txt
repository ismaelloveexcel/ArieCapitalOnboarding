🧭 App.jsx
import React, { useState } from "react";
import ProgressTabs from "./components/ProgressTabs";
import EntityInformation from "./pages/EntityInformation";
import GovernanceOwnership from "./pages/GovernanceOwnership";
import BusinessOperations from "./pages/BusinessOperations";
import AccountFinancialDetails from "./pages/AccountFinancialDetails";
import DocumentationCompliance from "./pages/DocumentationCompliance";

export default function App() {
  const [activeStep, setActiveStep] = useState(1);

  const steps = [
    "Entity Information",
    "Governance & Ownership",
    "Business Operations",
    "Account & Financial Details",
    "Documentation & Compliance",
  ];

  return (
    <div className="min-h-screen bg-gray-50 text-gray-800">
      <header className="border-b bg-white p-4 shadow-sm">
        <h1 className="text-2xl font-semibold text-blue-900">
          Corporate Account Application
        </h1>
        <p className="text-sm text-gray-500">
          All information is collected for due diligence and compliance purposes.
        </p>
      </header>

      <main className="p-6 max-w-7xl mx-auto">
        <ProgressTabs steps={steps} activeStep={activeStep} setActiveStep={setActiveStep} />

        <div className="mt-8">
          {activeStep === 1 && <EntityInformation />}
          {activeStep === 2 && <GovernanceOwnership />}
          {activeStep === 3 && <BusinessOperations />}
          {activeStep === 4 && <AccountFinancialDetails />}
          {activeStep === 5 && <DocumentationCompliance />}
        </div>
      </main>

      <footer className="border-t bg-white text-xs text-gray-500 text-center p-3 mt-8">
        Confidentiality Notice: Information provided in this application is strictly confidential
        and will be used only for regulatory and compliance verification.
      </footer>
    </div>
  );
}

🧩 ProgressTabs.jsx
export default function ProgressTabs({ steps, activeStep, setActiveStep }) {
  return (
    <nav className="flex justify-between bg-white rounded-md shadow-sm overflow-hidden">
      {steps.map((label, index) => (
        <button
          key={label}
          onClick={() => setActiveStep(index + 1)}
          className={`flex-1 px-4 py-3 text-sm border-r last:border-0 transition-all
            ${
              activeStep === index + 1
                ? "bg-blue-900 text-white font-medium"
                : "bg-gray-50 text-gray-600 hover:bg-gray-100"
            }`}
        >
          {`Step ${index + 1}`} — {label}
        </button>
      ))}
    </nav>
  );
}

✅ VerificationBadge.jsx
export default function VerificationBadge({ status }) {
  const colors = {
    Verified: "bg-green-100 text-green-700 border-green-400",
    Pending: "bg-yellow-100 text-yellow-700 border-yellow-400",
    Rejected: "bg-red-100 text-red-700 border-red-400",
  };
  return (
    <span
      className={`text-xs px-2 py-1 border rounded-full ${
        colors[status] || "bg-gray-100 text-gray-600 border-gray-300"
      }`}
    >
      {status || "Pending Verification"}
    </span>
  );
}
